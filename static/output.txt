SELECT 
  d.dept_name, 
  SUM(p.salary) AS total_salary
FROM 
  employee_tbl e 
  JOIN dept_tbl d ON e.dept_id = d.dept_id 
  JOIN payroll_data p ON e.emp_id = p.emp_id
GROUP BY 
  d.dept_name
ORDER BY 
  total_salary DESC;
